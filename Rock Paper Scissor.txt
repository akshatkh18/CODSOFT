import random

def ask_to_play_again():
    while True:
        choice = input("Do you want to play again? (yes/no): ").lower()
        if choice in ["yes", "no"]:
            return choice
        else:
            print("Invalid input. Please enter 'yes' or 'no'.")

def game():
    global computer_score 
    global player_score
    computer_score = 0
    player_score = 0
    play_again = "yes"

    while play_again == "yes":
        print(f"\nTURN {player_score + computer_score + 1}:\n")
        while True:
            user_choice = input("Enter your choice (rock, paper, or scissor): ").lower()
            if user_choice in ["rock", "paper", "scissor"]:
                break
            else:
                print("Invalid input. Please enter rock, paper, or scissor.")
                print("\n")

        computer_choice = random.choice(["rock", "paper", "scissor"])
        print("Computer's choice:", computer_choice)

        if user_choice == computer_choice:
            print("It's a tie!")
        elif (user_choice == "rock" and computer_choice == "scissor") or \
             (user_choice == "paper" and computer_choice == "rock") or \
             (user_choice == "scissor" and computer_choice == "paper"):
            print("You win!")
            player_score += 1
        else:
            print("You lose!")
            computer_score += 1

        print(f"Player's score: {player_score}")
        print(f"Computer's score: {computer_score}")
        print("\n")

        play_again = ask_to_play_again()

    # Display final scores and result
    print(f"\nFinal Player's Score: {player_score}")
    print(f"Final Computer's Score: {computer_score}")
    print("\n")

    if player_score > computer_score:
        print("Result: You win the game!")
        print("Congratulations!")
    elif player_score == computer_score:
        print("Result: It's a tie!")
        print("Try again!")
    else:
        print("Result: You lose the game!")

    print("\n\n")
    print("Thank you for playing!")
    print("See you next time!")

# Start the game
game()
