import tkinter as tk

class CalculatorApp:
    def __init__(self, root):
        # Initialize the main window
        self.root = root
        self.root.title("Calculator")
        self.root.geometry("400x500")
        self.root.configure(bg="black")

        # StringVar to hold the input/output
        self.expression = ""

        # Display widget
        self.display = tk.Entry(root, bg="gray20", fg="white", font=("Helvetica", 24), borderwidth=0, insertbackground="white")
        self.display.pack(expand=True, fill='both')

        # Button frame
        button_frame = tk.Frame(root, bg="black")
        button_frame.pack(expand=True, fill='both')

        # Button layout
        buttons = [
            '7', '8', '9', '/',
            '4', '5', '6', '*',
            '1', '2', '3', '-',
            'C', '0', '=', '+'
        ]

        row_val = 0
        col_val = 0
        for button in buttons:
            action = lambda x=button: self.on_button_click(x)
            tk.Button(button_frame, text=button, bg="gray30", fg="white", font=("Helvetica", 20), borderwidth=0, command=action).grid(row=row_val, column=col_val, sticky="nsew")

            col_val += 1
            if col_val > 3:
                col_val = 0
                row_val += 1

        # Configure button frame rows and columns to expand equally
        for i in range(4):
            button_frame.columnconfigure(i, weight=1)
            button_frame.rowconfigure(i, weight=1)

    def on_button_click(self, char):
        if char == "C":
            self.expression = ""
        elif char == "=":
            try:
                self.expression = str(eval(self.expression))
            except Exception as e:
                self.expression = "Error"
        else:
            self.expression += str(char)
        self.update_display()

    def update_display(self):
        self.display.delete(0, tk.END)
        self.display.insert(tk.END, self.expression)

if __name__ == "__main__":
    # Create the main window and run the application
    root = tk.Tk()
    app = CalculatorApp(root)
    root.mainloop()
